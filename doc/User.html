def password_complexity
  if password.present? and not password.match(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[\W]).+/)
    errors.add :password, "must include at least one of each: lowercase letter, uppercase letter, numeric digit, special character."
  end
end
